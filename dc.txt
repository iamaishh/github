FROM ubuntu:22.04
RUN apt update && apt install -y curl wget iputils-ping jq nginx unzip net-tools
COPY index.html /var/www/html/index.nginx-debian.html
ADD style.css /var/www/html
CMD ["nginx","-g","daemon off;"]


FROM tomcat:8.0
LABEL "Owner"="TOMCAT"
RUN cd /usr/local/tomcat/webapps && rm -rf *
COPY ROOT.war /usr/local/tomcat/webapps
CMD ["catalina.sh", "run"]


services:
  frontend:
    build:
      context: .
      dockerfile: netflixfile
    ports:
      - "8200:80"

  backend:
    build:
      context: .
      dockerfile: tomcatfile
    ports:
      - "8500:8080"
